### 获取n唯数组的最大深度

```c++
#include <bits/stdc++.h>
using namespace std;

int main() {
    int le = 0,mx = 0;
    char ch;
    while(cin >> ch) {
        if (ch == '[') le += 1;
        else if (ch == ']') le -= 1;
        if (mx < le) mx = le;
    }
    cout << mx << endl;
    
    return 0;
}
```

### 爬楼梯２

```c++
#include <bits/stdc++.h>
using namespace std;
int A[3][1000];
int main() {
    int n;
    cin >> n;
    A[0][0] = A[0][1] = A[1][0] = A[1][1] = A[2][0] = A[2][1]=1;
    
    for (int i = 3; i <= n; i++) {
        int ind = i%3;
        for (int j = 1; j <= A[(i - 1)%3][0];j++) {
            A[ind][j] += A[(i - 1) % 3][j];
        }
        A[ind][0] = A[(i - 1) % 3][0];
        for (int j = 1; j <= A[ind][0]; j++) {
            if (A[ind][j] < 10) continue;
            A[ind][j + 1] += A[ind][j] /10;
            A[ind][j] = A[ind][j]% 10;
            if (j == A[ind][0]) A[ind][0] += 1;
        }
       
    }
    for (int i = A[n%3][0]; i >= 1; i--) {
        cout << A[n%3][i];
    }
    cout << endl;
    return 0;
}
```
